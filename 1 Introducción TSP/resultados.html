def t_ejecucion(n):
    X = []
    for i in range (n):
        X.append(np.random.rand(2))
    X = np.array(X)
    t0 = time()    
    tsp.TSP(X)
    t1 = time()
    return t1-t0

def t_2an(n):
    times = []
    for n in range(2,10):
        times.append(t_ejecucion(n))
        #print(f'{n}-> {times[-1]}')
    return times

def t_promedio(n, veces):
    times = []
    for n in range(veces):
        times.append(t_2an(n))
    times = np.array(times)
    return times.mean(axis=0)
